find_package(Threads REQUIRED)

find_package(PostgreSQL)
set_package_properties(PostgreSQL PROPERTIES URL "https://www.postgresql.org/" DESCRIPTION "PostgresSQL support" TYPE OPTIONAL)

if (POSTGRESQL_FOUND)
    add_library(sqlpp20-connector-postgresql INTERFACE)

   target_include_directories(sqlpp20-connector-postgresql INTERFACE
        ${PostgreSQL_INCLUDE_DIRS}
        $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/include>
        $<INSTALL_INTERFACE:include>
    )

    target_link_libraries(sqlpp20-connector-postgresql INTERFACE sqlpp20)
    target_link_libraries(sqlpp20-connector-postgresql INTERFACE "${PostgreSQL_LIBRARIES}")

    add_subdirectory(tests)

    install(DIRECTORY "${CMAKE_CURRENT_SOURCE_DIR}/include/sqlpp20"
            DESTINATION include
            FILES_MATCHING PATTERN "*.h")
endif()
